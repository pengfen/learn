$('.selector').click(function() { var form = $('#uploadForm'); $(this).attr('disabled', 'true'); form.submit(); }); 去除layer弹框流动条 $('.layui-layer-content').css('overflow-y', 'hidden'); curl 上传文件 (调用第三方接口) /** * $url 第三方接口地址 * $allowType 上传文件类型 * txt 文件类型 text/plain * excel 文件类型 application/vnd.openxmlformats-officedocument.spreadsheetml.sheet,application/vnd.ms-excel * 图片文件类型 'image/png', 'image/gif', 'image/jpeg' */ function upload($url, $allowType){ // 设置相关参数 $result = array('code' => 0, 'message' => 'ok'); ini_set('upload_max_filesize', '20M'); ini_set('post_max_size', '20M'); ini_set('memory_limit', '128M'); $systemId = $this->get_system_id(); $userId = Yii::$app->session['userInfo']['userBase']['uid']; $file = $_FILES['file']; //3. 验证 文件类型 if (!empty($allowType)){ if (!in_array($file['type'], $allowType)){ $return['info'] = "文件类型不合法"; return $return; } } if ($file['error'] != 0) { //输出错误原因 switch ($file['error']){ case 1: $info = "文件大小超过了php.ini中的限制！"; break; case 2: $info = "文件大小超过了表单中的限制！"; break; case 3: $info = "文件只被部分上传!"; break; case 4: $info = "文件没有上传!"; break; case 6: $info = "找不到临时目录!"; break; case 7: $info = "文件写入失败！"; break; } $return['info'] = $info; return $return; } else { $filename = $file['name']; $tmpfile = $file['tmp_name']; $filetype = $file['type']; $data = $this->upload_file($url, $filename, $tmpfile, $filetype, $systemId, $userId); return $data; } } /** * curl上传文件 * * @param $url 目标主机地址 * @param $filename 上传文件名 * @param $path 上传临时目录 * @param $type 上传文件类型 */ function upload_file($url, $filename, $path, $type, $systemId, $userId){ //php 5.5以上的用法 if (class_exists('\CURLFile')) { $data = array('systemId' => $systemId, 'userId' => $userId, 'file' => new \CURLFile(realpath($path),$type,$filename)); } else { $data = array( 'file'=>'@'.realpath($path).";type=".$type.";filename=".$filename ); } $ch = curl_init(); curl_setopt($ch, CURLOPT_URL, $url); curl_setopt($ch, CURLOPT_POST, true ); curl_setopt($ch, CURLOPT_POSTFIELDS, $data); curl_setopt($ch, CURLOPT_HEADER, false); curl_setopt($ch, CURLOPT_RETURNTRANSFER, true); $return_data = curl_exec($ch); curl_close($ch); @unlink($path); return json_decode($return_data, true); }
