// 根据cid搜索 $('.cid').click(function() { var cid = $(this).prev('input').val(); if (cid) { if (cid <= 0 || isNaN(cid)) { layer.msg('客户ID必须是数字'); return false; } } var form = $('#cidForm'); form.submit(); }) 非表单回车提交
